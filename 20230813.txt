  public void actionPerformed(ActionEvent e) {
    if (e.getSource() == mainPageButton[0]) {
      mainFrame.remove(mainPage);
      mainFrame.add(page_A[1]);
      mainFrame.add(pageMenuBar_A, BorderLayout.NORTH);
      mainFrame.revalidate();
      mainFrame.repaint();
      pageNo_A.setSelectedIndex(1);
    }
    if (e.getSource() == mainPageButton[1]) {
      mainFrame.remove(mainPage);
      mainFrame.add(page_B[1]);
      mainFrame.add(pageMenuBar_B, BorderLayout.NORTH);
      mainFrame.revalidate();
      mainFrame.repaint();
      pageNo_B.setSelectedIndex(1);
    }
    if (e.getSource() == mainPageButton[2]) {
      mainFrame.remove(mainPage);
      mainFrame.add(settings);
      mainFrame.add(settingsMenuBar, BorderLayout.NORTH);
      mainFrame.revalidate();
      mainFrame.repaint();
    }
    if (e.getSource() == mainPageButton[3]) {
      System.exit(0);
    }
    if (e.getSource() == homePageButton_A && mainPage.getParent() == null) {
      mainFrame.remove(pageMenuBar_A);
      for (int i = 1; i <= 99; i++) {
        mainFrame.remove(page_A[i]);

  @Override
  public void mouseEntered(MouseEvent e) {
	  if(e.getSource()==flag[0]) {
		  EU_clip.setMicrosecondPosition(0);
		  EU_clip.start();
	  }
	  if(e.getSource()==flag[1]) {			  
		  tarantellaNapoletanaClip.setMicrosecondPosition(0);
		  tarantellaNapoletanaClip.start();  
	  }
	  if(e.getSource()==flag[2]) {
		  czerwoneJagodyClip.setMicrosecondPosition(0);
		  czerwoneJagodyClip.start();
	  }
	  for(int i=0;i<3;i++) {
		  if(e.getSource()==flag[i]) {
			  mainFrame.setCursor(Cursor.getPredefinedCursor(Cursor.CROSSHAIR_CURSOR));
		  }
	  }
    for (int i = 1; i <= 116; i++) {
      if (e.getSource() == PL_wyraz_A[i]) {
    	PL_wyraz_A[i].setBackground(gold);
        PL_wyraz_A[i].setBorder(BorderFactory.createLoweredSoftBevelBorder());       
        PL_clip_A[i].setMicrosecondPosition(0);
        PL_clip_A[i].start();
      }
      if (e.getSource() == IT_espr_A[i]) {
    	IT_espr_A[i].setBackground(gold);
        IT_espr_A[i].setBorder(BorderFactory.createLoweredSoftBevelBorder());
        IT_clip_A[i].setMicrosecondPosition(0);
        IT_clip_A[i].start();      
      }
    }
    for (int i = 1; i <= 180; i++) {
      if (e.getSource() == PL_wyraz_B[i]) {
    	PL_wyraz_B[i].setBackground(gold);
        PL_wyraz_B[i].setBorder(BorderFactory.createLoweredBevelBorder());
        PL_clip_B[i].setMicrosecondPosition(0);
        PL_clip_B[i].start();
      }
      if (e.getSource() == IT_espr_B[i]) {
    	IT_espr_B[i].setBackground(gold);
        IT_espr_B[i].setBorder(BorderFactory.createLoweredBevelBorder());
        IT_clip_B[i].start();
        IT_clip_B[i].setMicrosecondPosition(0);
      }
    }
  }
  public void mouseExited(MouseEvent e) {
	  if(e.getSource()==flag[0]) {	  
		  EU_clip.stop();
	  }
	  if(e.getSource()==flag[1]) {	  
		  tarantellaNapoletanaClip.stop();
	  }
	  if(e.getSource()==flag[2]) {	  
		  czerwoneJagodyClip.stop();
	  }
	  for(int i=0;i<3;i++) {
		  if(e.getSource()==flag[i]) {
			  mainFrame.setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
		  }
	  }
    for (int i = 1; i <= 116; i++) {
      if (e.getSource() == PL_wyraz_A[i]) {
    	PL_wyraz_A[i].setBackground(null);
        PL_wyraz_A[i].setBorder(BorderFactory.createRaisedSoftBevelBorder());
        PL_clip_A[i].stop();
      }
      if (e.getSource() == IT_espr_A[i]) {
    	IT_espr_A[i].setBackground(null);
        IT_espr_A[i].setBorder(BorderFactory.createRaisedSoftBevelBorder());       
        IT_clip_A[i].stop();
      }
    }
    for (int i = 1; i <= 180; i++) {
      if (e.getSource() == PL_wyraz_B[i]) {
    	PL_wyraz_B[i].setBackground(null);
        PL_wyraz_B[i].setBorder(BorderFactory.createRaisedBevelBorder());      
        PL_clip_B[i].stop();
      }
      if (e.getSource() == IT_espr_B[i]) {
        IT_espr_B[i].setBackground(null);
        IT_espr_B[i].setBorder(BorderFactory.createRaisedBevelBorder());      
        IT_clip_B[i].stop();
